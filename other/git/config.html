# Git 配置

# 查看配置列表
git config --list
# 查看全局配置
git config --global http.proxy
# 添加全局配置
git config --global user.name "Your Name"
git config --global user.email "Your Email"
git config --global http.proxy 'http://127.0.0.1:1080'
# 删除全局配置
git config --global --unset http.proxy
